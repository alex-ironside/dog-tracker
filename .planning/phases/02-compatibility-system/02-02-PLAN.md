    ```
    import type { CompatibilityStatus, CompatibilityEntry } from '@/types'

    export type ConflictingPair = { idA: string; idB: string; status: 'conflict' | 'unknown' }

    export function pairKey(idA: string, idB: string): string {
      return [idA, idB].sort().join('|')
    }

    export function buildCompatMap(entries: CompatibilityEntry[]): Map<string, CompatibilityStatus> {
      return new Map(entries.map(e => [pairKey(e.dogIdA, e.dogIdB), e.status]))
    }
    ```

    scoreGroup implementation:
    - `const n = dogIds.length`
    - `const totalPairs = (n * (n - 1)) / 2`
    - If `totalPairs === 0` return `100`
    - Weights: `{ compatible: 1.0, neutral: 0.5, unknown: 0.25, conflict: 0.0 }`
    - Nested loop `i=0..n-1, j=i+1..n-1`, lookup `compatMap.get(pairKey(dogIds[i], dogIds[j])) ?? 'unknown'` (D-03: missing = unknown)
    - Sum weights, return `Math.round((sum / totalPairs) * 100)`

    getConflictsInGroup implementation:
    - Same nested loop over all pairs
    - Lookup status: `compatMap.get(pairKey(dogIds[i], dogIds[j])) ?? 'unknown'`
    - If status is `'conflict'` or `'unknown'`, push `{ idA: dogIds[i], idB: dogIds[j], status }` to result
    - Return result array
