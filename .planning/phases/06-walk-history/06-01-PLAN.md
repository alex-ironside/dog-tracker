    Update `migrate` function to handle v1 -> v2:
    ```typescript
    export const CURRENT_SCHEMA_VERSION = 2

    export function migrate(persistedState: unknown, version: number): AppState {
      const state = persistedState as AppState

      if (version < 2) {
        return {
          ...state,
          walkHistory: (state as any).walkHistory ?? [],
          schemaVersion: 2,
        }
      }

      return { ...state, schemaVersion: CURRENT_SCHEMA_VERSION }
    }
    ```

    The `(state as any).walkHistory ?? []` ensures that even if walkHistory is somehow already present (defensive), it is preserved; otherwise it defaults to empty array.
